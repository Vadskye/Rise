    \IfNoValueTF{#5}
    {\spelltgt{#3}[#2]}

\DeclareDocumentEnvironment{spellsection}{o m o}
{
    \vspace{1em}
    \begin{thesamepage}
    \setlength\multicolsep{0pt}
    \begin{multicols}{2}
        \IfValueTF{#1}
        {
            \lowercase{\hypertarget{spell:#1~#2}{}}
            \label{spell:#1~#2}
            \hypertarget{spell:#1~#2}
                {\subsubsection{#2,~#1}}
        }
        {
            \lowercase{\hypertarget{spell:#2}{}}
            \label{spell:#2}
            \hypertarget{spell:#2}
                {\subsubsection{#2}}
        }
        \IfValueTF{#3}
        {
            \columnbreak
            \begin{flushright}
                \large\textbf{\nth{#3}~Level}
            \end{flushright}
        }
        {
            \columnbreak
            \begin{flushright}
            \end{flushright}
        }
    \end{multicols}
}
{
    \end{thesamepage}
}

    {
        \parhead{Body~Location} #4
    }

    \parhead{Aura,~Spellpower} \magicitemaurastrength{#3}~#5,~\magicitemspellpower*{#3}

    \IfValueT{#6}
    {
        \parhead{Activation} #6
    }
    \par
}
