        )
    \fi
}

    % category of item level the item falls into

            #1 <= 50 ? 1 : (
                #1 <= 100 ? 2 : (
                    #1 <= 250 ? 3 : (
                        #1 <= 500 ? 4 : (
                            #1 <= 800 ? 5 : (
                                #1 <= 1200 ? 6 : (
                                    #1 <= 1800 ? 7 : (
                                        #1 <= 2750 ? 8 : (
                                            #1 <= 4000 ? 9 : (
                                                #1 <= 6500 ? 10 : (
                                                    #1 <= 10000 ? 11 : 12
                                                )
                                            )
                                        )
                                    )
                                )
                            )
                        )
                    )
                )
            )
        )
    }
    \ifstr{\pgfmathresult}{0}{1/2}[\nth{\pgfmathresult}]
}
